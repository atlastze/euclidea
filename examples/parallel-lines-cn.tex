\documentclass{standalone}
\usepackage{ctex}
\usepackage{fontset}
\usepackage{tikz}

\usetikzlibrary{euclidea,angles}

\begin{document}

\begin{tikzpicture}
  \node (A) at (-2,0) [red,left] {$A$};
  \node (B) at ( 3,.5)[red,right] {$B$};
  \node (C) at (-2,2) [blue,left] {$C$};
  \coordinate [translate={A,B,C}] (D);
  \node [blue,right] at (D) {$D$};
  \node (E) at (60:-5mm) [below] {$E$};
  \node (F) at (60:3.5cm) [above] {$F$};
  \coordinate [intersect={A,B,E,F}] (X);
  \coordinate [intersect={C,D,E,F}] (Y);

  \path
    (A) edge [red,thick] (B)
    (C) edge [blue,thick] (D)
    (E) edge [thick] (F)
      pic ["$\alpha$", draw, fill=yellow] {angle = F--X--A}
      pic ["$\beta$", draw, fill=green!30] {angle = B--X--F}
      pic ["$\gamma$", draw, fill=yellow] {angle = E--Y--D}
      pic ["$\delta$", draw, fill=green!30] {angle = C--Y--E};

    \node at ($(D)!.5!(B)$) [
      right=1cm,text width=6cm,
      rounded corners,fill=red!20,inner sep=1ex,
      font=\CJKfamily{楷体}
    ]{
      如果直线 $\color{red}AB$ 与直线 $\color{blue}CD$ 平行时, 也即, ${\color{red}AB} \mathbin{\parallel} \color{blue}CD$, 那么有 $\alpha = \gamma$ 和 $\beta = \delta$.
      简写为: {\color{red}{两直线平行, 内错角相等.}}
    };
\end{tikzpicture}

\end{document}
