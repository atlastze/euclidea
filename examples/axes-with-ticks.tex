\documentclass[10pt]{standalone}
\usepackage{tikz} 

\begin{document}
  \begin{tikzpicture}
  %画x和y轴坐标
  \draw[-latex] (-5.2,0)--(5.2,0);
  \draw[-latex] (0,-5.2)--(0,5.2);
  %画刻度
  \foreach \x in {0,1,...,8}
  {
    \draw[xshift=\x cm] (-4,0) -- (-4,0.1);
    \draw[yshift=\x cm] (0,-4) -- (0.1,-4);
  };  
  %标坐标原点
  \node[below] at (0.2,0){0};
  %标x轴刻度值
  \foreach \x in {-4,-3,...,-1}
    \node[below] at(\x,0){\x};
  \foreach \y in {1,2,...,4}
    \node[below] at(\y,0){\y};
  % 标注y轴刻度
  \foreach \y in {-4,-3,...,-1}
    \node[left] at(0,\y){\y};
  \foreach \y in {1,2,...,4}
    \node[left] at(0,\y){\y};
  \end{tikzpicture}
\end{document}
