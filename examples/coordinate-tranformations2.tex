% For this you need to remember that TikZ composes transformations in the opposite 
% order than the naturally expected one (i.e. it acts on the coordinate plane (canvas), 
% not on the objects).

\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{euclidea}

\begin{document}
\begin{tikzpicture}[scale = 0.5]
  \draw[help lines] (0,0) grid (5,5);
  \draw [-latex](-0.5,0) -- (5.5,0) node[below] {$x$}; %x-axis
  \draw [-latex](0,-0.5) -- (0,5.5) node[left] {$y$}; %y-axis

  \draw (2,1) circle (2pt) node[below] {$A$};
  \draw [shift={(1,2)}, red] (2,1) circle (2pt) node[below] {$B$};
  \draw [rotate around={45:(1,1)}, green] (2,1) circle (2pt) node[below] {$C$};
  \draw [shift={(1,2)}, rotate around={45:(1,1)}, magenta] (2,1) circle (2pt) node[below] {$D$};
  \draw [rotate around={45:(1,1)}, shift={(1,2)}, teal] (2,1) circle (2pt) node[below] {$E$};
  \draw [purple,cm={cos(45), sin(45), -sin(45), cos(45), (1,2)}] (2,1) circle (2pt) node[below] {$F$};
\end{tikzpicture}
\end{document}
