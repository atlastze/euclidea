\documentclass{standalone}

\usepackage{tikz}
\usetikzlibrary{calc}

\newcommand\centerarc{} % just for safety
\def\centerarc[#1]#2(#3)#4(#5:#6:#7)% [draw options] (center) (initial angle:final angle:radius)
  {\draw[#1]($(#3)+({#7*cos(#5)},{#7*sin(#5)})$)arc(#5:#6:#7);}
  
\begin{document}
  \begin{tikzpicture}
    \draw[help lines] (0,0) grid (10,10);
    \centerarc[thick] (1,2) (0:50:5);
    \coordinate (O) at (1,2);
    \centerarc[thick,red] (O) (60:60-40:5);
  \end{tikzpicture}
\end{document}
