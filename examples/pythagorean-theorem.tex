\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{euclidea}

\begin{document}
\begin{tikzpicture}
  \coordinate (A) at (0,3);
  \coordinate (B) at (4,0);
  \coordinate (C) at (0,0);
  \coordinate[intercept/length={B,C},
    intercept/scale=-1,intercept={A,C}] (D);
  \coordinate[intercept/length={A,C},
    intercept/scale=-1,intercept={B,C}] (E);
  \coordinate[revolve/angle=90,revolve={A,B}] (F);
  \coordinate[revolve/angle=-90,revolve={B,A}] (G);
  \coordinate[intersect={D,F,E,G}] (H);

  \fill[cyan] (A) -- (B) -- (C)
              (D) -- (A) -- (F)
              (B) -- (E) -- (G)
              (G) -- (H) -- (F);

  \foreach \p/\placement in {A/left, B/below, C/below left,
  D/above left, E/below right, F/above, G/right, H/above right}{
    \fill[red] (\p) circle (2pt);
    \draw (\p) node[\placement] {$\p$};
  }

\end{tikzpicture}
\end{document}
