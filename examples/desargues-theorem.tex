\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{euclidea}

\begin{document}
\begin{tikzpicture}
  \coordinate (O) at (0,5);
  \coordinate (A1) at (-1,2.5);
  \coordinate (B1) at (2,1.5);
  \coordinate (C1) at (0,3.5);
  \coordinate (A2) at ($(O)!3.5!(A1)$);
  \coordinate (B2) at ($(O)!2!(B1)$);
  \coordinate (C2) at ($(O)!7!(C1)$);
  
  \coordinate[intersect={B1,C1,B2,C2}] (D);
  \coordinate[intersect={C1,A1,C2,A2}] (E);
  \coordinate[intersect={A1,B1,A2,B2}] (F);

  \fill[cyan] (A1) -- (B1) -- (C1) -- cycle;
  \fill[green] (A2) -- (B2) -- (C2) -- cycle;

  \draw (D) -- (C1) (D) -- (C2)
        (E) -- (C1) (E) -- (C2)
        (F) -- (A1) (F) -- (A2);

  \draw[end modifier=1.2,parallel={O,A2,O}];
  \draw[end modifier=1.2,parallel={O,B2,O}];
  \draw[end modifier=1.2,parallel={O,C2,O}];
  \draw[start modifier=-.05,end modifier=1.05,parallel={E,F,E}];
    
  \foreach \p/\placement in {O/above,
  A1/above left,B1/above right,C1/above,
  A2/below left,B2/below right,C2/below,
  D/above,E/above,F/above}{
    \fill[red] (\p) circle (2pt);
    \draw (\p) node[\placement] {$\p$};
  }
\end{tikzpicture}
\end{document}
